#include <stdio.h>

int main()
{
	int score[5], max=0, min=10, tot;
	double avg;
	
	printf("5명 심사위원의 점수 입력 : ");
	scanf("%d %d %d %d %d",&score[0],&score[1],&score[2],&score[3],&score[4]);
	
	for(int i=0;i<5;i++)
	{
		if(score[i]>max)
			max=score[i];
	}
	
	for(int i=0;i<5;i++)
	{
		if(score[i]<min)
			min=score[i];
	}
	
	printf("유효점수 : ");

	for(int i=0;i<5;i++)
	{
		if((score[i]!=max)&&(score[i]!=min))
		{
			printf("  %d",score[i]);
			
			tot+=score[i];
		}
	}

	printf("\n평균 : %.1lf",(double)tot/3);
	
	return 0;
}
